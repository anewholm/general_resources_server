00366     byte_t * grow( size_t size ) // throw(std::bad_alloc)
00367     {
00368         // write the cached value of current into the node.
00369         // OK to do this even if later statements throw.
00370         m_current_node->m_head.m_current = m_current;
00371 
00372         // Do we have a node with available memory already?
00373         if( m_current_node->m_head.m_next )
00374         {
00375             // Does this node have enough room?
00376             if( size <= m_current_node->m_head.m_next->size() )
00377             {
00378                 m_current_node  = m_current_node->m_head.m_next;
00379                 m_current       = m_current_node->m_head.m_current = m_current_node->m_mem + size;
00380                 m_end           = m_current_node->m_head.m_end;
00381                 return m_begin  = m_current_node->m_mem;
00382             }
00383 
00384             // Create a new node and insert it into the list
00385             stack_node * new_node = static_cast<stack_node*>(
00386                 ::operator new( size + offsetof( stack_node, m_mem ) ) );
00387 
00388             new_node->m_head.m_back = m_current_node;
00389             new_node->m_head.m_next = m_current_node->m_head.m_next;
00390 
00391             m_current = m_end = new_node->m_head.m_current =
00392                 new_node->m_head.m_end = new_node->m_mem + size;
00393 
00394             m_current_node->m_head.m_next->m_head.m_back = new_node;
00395             m_current_node->m_head.m_next = new_node;
00396             m_current_node = new_node;
00397 
00398             return m_begin = m_current_node->m_mem;
00399         }
00400 
00401         // We need to create a new node from scratch
00402         size_t new_size = detail::regex_max( size,
00403             static_cast<size_t>(DYNAMIC_BLOCK_SIZE) - offsetof( stack_node, m_mem ) );
00404 
00405         stack_node * new_node = static_cast<stack_node*>(
00406             ::operator new( new_size + offsetof( stack_node, m_mem ) ) );
00407 
00408         new_node->m_head.m_back = m_current_node;
00409         new_node->m_head.m_next = 0;
00410 
00411         m_current = new_node->m_head.m_current = new_node->m_mem + size;
00412         m_end     = new_node->m_head.m_end     = new_node->m_mem + new_size;
00413 
00414         m_current_node->m_head.m_next = new_node;
00415         m_current_node = new_node;
00416 
00417         return m_begin = m_current_node->m_mem;
00418     }

00420     byte_t * allocate( size_t size ) // throw(std::bad_alloc)
00421     {
00422         // This is the ptr to return
00423         byte_t * mem = m_current;
00424 
00425         // Advance the high-water mark
00426         m_current += size;
00427 
00428         // Check to see if we have overflowed this buffer
00429         if( std::less<void*>()( m_end, m_current ) ) // if( m_end < m_current )
00430         {
00431             // oops, back this out.
00432             m_current = mem;
00433 
00434             // allocate a new block and return a ptr to the new memory
00435             return grow( size );
00436         }
00437 
00438         return mem;
00439     }

00572     template< typename T >
00573     inline void push( T const & t ) // throw(std::bad_alloc,...)
00574     {
00575         // Make sure that the alignment for type T is not worse
00576         // than our declared alignment.
00577         detail::static_assert<( AlignmentT >= detail::alignof<T>::value )> const align_test;
00578         static_cast<void>(align_test);
00579 
00580         // If T won't throw in copy c'tor then we don't need to use an unwinder object.
00581         typedef typename detail::select< AssumePodT || detail::has_trivial_copy<T>::value,
00582             dummy_unwinder, real_unwinder >::type unwinder;
00583 
00584         // If this throws, it doesn't change state,
00585         // so there is nothing to roll back.
00586         byte_t * pb = allocate( aligned_sizeof<T>::with_rtti );
00587 
00588         // Rolls back the allocate if later steps throw
00589         // BUGBUG we can do the alloc, but not update m_current until after
00590         // the copy c'tor to avoid the need for an unwinder object
00591         unwinder guard( this, aligned_sizeof<T>::with_rtti );
00592 
00593         new ( pb ) T( t ); // Could throw if ! has_trivial_copy<T>::value
00594 
00595         // If we are debugging the stack, then push a pointer to the type_info
00596         // for this type T. It will be checked in pop().
00597         if( RuntimeTypeCheckT )
00598         {
00599             detail::to_type<vtable_ptr>( pb + aligned_sizeof<T>::no_rtti ) = & detail::type_info_ex<T,AlignmentT>::vtable;
00600         }
00601 
00602         // ok, everything succeeded -- dismiss the guard
00603         guard.dismiss();
00604     }

00608     {
00609         detail::static_assert<( AlignmentT >= detail::alignof<T>::value )> const align_test;
00610         static_cast<void>(align_test);

00616             byte_t * pti = m_current - aligned_sizeof<vtable_ptr>::no_rtti;

00791     // things to and from the stack. This is especially useful
00792     // if you want to push a bunch of things "atomically".  For
00793     // instance:
00794     //
00795     // typedef hetero_stack<>::scoped_pop scoped_pop;
00796     // scoped_pop p1 = stack.scoped_push( int(1) ); // could throw
00797     // scoped_pop p2 = stack.scoped_push( std::string("foo") ); // could throw
00798     // stack.push( float(3.14159) ); // could throw
00799     // p2.dismiss(); // ok, nothing threw, so ...
00800     // p1.dismiss(); //  ... dismiss the scoped_pops

00802     // If p2 and p1 are not dismissed, as in the case when an
00803     // exception gets thrown, then they automatically pop their
00804     // arguments from the stack.

00806     class scoped_pop_base
00807     {
00808         scoped_pop_base & operator=( scoped_pop_base const & ); // disallow assignment
00809     protected:

00813           : m_pstack( pstack )
00814         {
00815         }
00816         scoped_pop_base( scoped_pop_base const & right ) // throw() // destructive copy
00817           : m_pstack( right.m_pstack )
00818         {
00819             right.dismiss();
00820         }
00821     public:
00822         void dismiss() const // throw()
00823         {

00825         }
00826     };
