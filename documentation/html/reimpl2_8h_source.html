00334 template< typename AllocT >
00335 inline void pool_impl<AllocT>::new_block( size_t size )
00336 {
00337     size_t blocksize = regex_max( m_data.m_default_size, size ) + offsetof( mem_block, m_data );
00338     mem_block * pnew = reinterpret_cast<mem_block*>( m_data.get_allocator().allocate( blocksize, 0 ) );
00339     if( 0 == pnew )
00340     {
00341         throw std::bad_alloc();
00342     }
00343     pnew->m_offset      = 0;
00344     pnew->m_blocksize   = blocksize;
00345     pnew->m_pnext       = m_data.m_pfirst;
00346     m_data.m_pfirst     = pnew;
00347 }

00349 template< typename AllocT >
00350 inline void * pool_impl<AllocT>::allocate( size_t size )
00351 {
00352     if( 0 == size )
00353         size = 1;
00354 
00355     if( 0 == m_data.m_pfirst || m_data.m_pfirst->m_offset + size > m_data.m_default_size )
00356         new_block( size );
00357 
00358     void * pnew = m_data.m_pfirst->m_data + m_data.m_pfirst->m_offset;
00359 
00360     // ensure returned pointers are always suitably aligned
00361     m_data.m_pfirst->m_offset += ( ( size + alignof<not_pod>::value - 1 )
00362                                  & ~( alignof<not_pod>::value - 1 ) );
00363 
00364     return pnew;

00768     (
00769         std::vector<backref_type,Alloc1T> & rgbackrefs,
00770         std::list<size_t,Alloc2T> const & invisible
00771     )
00772     {
00773         typedef typename std::list<size_t,Alloc2T>::const_iterator iter_type;
00774 
00775         // Remove information about the "invisible" groups
00776         if( rgbackrefs[0].matched )
00777         {
00778             size_t dropped = 0;
00779             iter_type const end = invisible.end();
00780             iter_type curr = invisible.begin(), next = invisible.begin();
00781 
00782             for( ; end != curr; curr = next, ++dropped )
00783             {
00784                 if( end == ++next )

00786                     std::copy(

00791                 else

00793                     std::copy(
00794                         rgbackrefs.begin() + *curr + 1,
00795                         rgbackrefs.begin() + *next,
00796                         rgbackrefs.begin() + *curr - dropped );
00797                 }
00798             }
00799 
00800             rgbackrefs.resize( rgbackrefs.size() - dropped );
00801         }
00802         else
00803         {

00851     template< typename AllocT >
00852     static bool _do_match
00853     (
00854         rpattern_type const & pat,

00856         IterT ibegin,
00857         IterT iend,
00858         bool use_null
00859     )
00860     {
00861         typedef typename basic_match_results<IterT,AllocT>::backref_vector backref_vector;
00862 
00863         results.m_ibegin = ibegin;
00864         match_param<IterT> param( ibegin, ibegin, iend, 0, 0 );
00865 
00866         if( GLOBAL & pat.flags() ) // do a global find
00867         {
00868             // The NOBACKREFS flag is ignored in the match method.
00869             bool const fAll   = ( ALLBACKREFS   == ( ALLBACKREFS   & pat.flags() ) );
00870             bool const fFirst = ( FIRSTBACKREFS == ( FIRSTBACKREFS & pat.flags() ) );
00871 
00872             backref_vector rgtempbackrefs( results.m_rgbackrefs.get_allocator() );

00876                 backref_type const & br = param.m_prgbackrefs[0];
00877

00879                 if( fFirst )
00880                 {
00881                     rgtempbackrefs.push_back( br );
00882                 }
00883                 else if( fAll )
00884                 {
00885                     rgtempbackrefs.insert(
00886                         rgtempbackrefs.end(),
00887                         results.m_rgbackrefs.begin(),

00896                 param.m_no0len = ( br.first == br.second );

00899             // restore the backref vectors
00900             results.m_rgbackrefs.swap( rgtempbackrefs );
00901             return ! results.m_rgbackrefs.empty();

00958     template< typename CharT, typename TraitsT, typename AllocT >
00959     static size_t _do_split
00960     (
00961         rpattern_type const & pat,
00962         basic_split_results<CharT, TraitsT, AllocT> & results,
00963         IterT ibegin,
00964         IterT iend,
00965         int limit,
00966         bool use_null
00967     )
00968     {
00969         typedef typename basic_split_results<CharT, TraitsT, AllocT>::string_type   string_type;
00970         typedef typename rebind<AllocT, backref_type>::type                         backref_allocator;
00971 
00972         std::vector<backref_type,backref_allocator> rgbackrefs( 
00973             convert_allocator<backref_type>( results.strings().get_allocator(), 0 ) );
00974             
00975         typedef typename rebind<AllocT, CharT>::type                                char_allocator_type;
00976         char_allocator_type char_allocator = 
00977             convert_allocator<CharT>( results.strings().get_allocator(), 0 );
00978 
00979         // reserve some initial space
00980         results.strings().clear();
00981         results.strings().reserve( 10 );
00982 
00983         match_param<IterT> param( ibegin, ibegin, iend, 0, 0 );
00984 
00985         while( 1 != limit && _do_try_match( pat, param, rgbackrefs, use_null ) )
00986         {
00987             backref_type const & br = param.m_prgbackrefs[0];
00988             param.m_no0len = ( br.first == br.second );
00989 
00990             // discard zero-width matches at the beginning and end of the buffer
00991             if( param.m_no0len )
00992             {
00993                 // if we're at the beginning, skip
00994                 if( br.first == param.m_ibufferbegin )
00995                     continue;
00996 
00997                 // if we're at the end, break
00998                 if( use_null ? 0 == *param.m_imatchbegin : param.m_imatchbegin == param.m_iend )
00999                     break;
01000             }
01001 
01002             string_type tmp( param.m_imatchbegin, br.first, char_allocator );
01003             results.strings().push_back( tmp );
01004             param.m_imatchbegin = br.second;
01005 
01006             // add any groups
01007             for( size_t i = 1; i < rgbackrefs.size(); ++i )
01008             {
01009                 backref_type const & br = rgbackrefs[i];
01010                 string_type tmp( br.first, br.second, char_allocator );
01011                 results.strings().push_back( tmp );
01012             }
01013 
01014             if( limit > 0 )
01015                 --limit;
01016         }
01017 
01018         // append the last string, unless it's empty and limit is 0
01019         if( use_null )
01020         {
01021             if( *param.m_imatchbegin || 0 != limit )
01022                 results.strings().push_back( string_type( &*param.m_imatchbegin, char_allocator ) );
01023         }
01024         else
01025         {
01026             if( param.m_imatchbegin != param.m_iend || 0 != limit )
01027                 results.strings().push_back( string_type( param.m_imatchbegin, param.m_iend, char_allocator ) );
01028         }
01029 
01030         // remove trailing empty fields
01031         if( 0 == limit )
01032         {
01033             while( results.size() && results.back().empty() )
01034             {
01035                 results.strings().pop_back();
01036             }
01037         }
01038 
01039         return results.size();
01040     }

01042     template< typename CharT, typename TraitsT, typename AllocT >
01043     static size_t _do_subst_internal
01044     (
01045         std::basic_string<CharT, TraitsT, AllocT> & str,
01046         basic_subst_results<CharT, TraitsT, AllocT> const & results,
01047         rpattern_type const & pat,
01048         size_type strpos,
01049         size_type strlen
01050     )
01051     {
01052         typedef subst_list_type::const_iterator iter_type;
01053         enum { UPPER = -1, NIL, LOWER } next = NIL, rest = NIL;
01054         bool first = true;
01055         size_t old_strpos = strpos;
01056         typename std::basic_string<CharT, TraitsT, AllocT>::iterator itstrlen = str.begin();
01057         std::advance( itstrlen, strpos + strlen );
01058         std::basic_string<char_type> const & subst = pat.get_subst();
01059 
01060         for( iter_type isubst = pat.m_subst_list.begin(); pat.m_subst_list.end() != isubst; ++isubst )
01061         {
01062             size_t sublen = 0;
01063             typename std::basic_string<CharT, TraitsT, AllocT>::const_iterator itsubpos1; // iter into str
01064             typename std::basic_string<CharT, TraitsT, AllocT>::const_iterator itsublen1;
01065             typename std::basic_string<char_type>::const_iterator              itsubpos2; // iter into subst string
01066             typename std::basic_string<char_type>::const_iterator              itsublen2;
01067             typename std::basic_string<CharT, TraitsT, AllocT>::iterator       itstrpos = str.begin();
01068             std::advance( itstrpos, strpos );
01069 
01070             switch( isubst->m_stype )
01071             {
01072             case subst_node::SUBST_STRING:
01073                 itsubpos2 = subst.begin();
01074                 std::advance( itsubpos2, isubst->m_subst_string.m_rstart );
01075                 itsublen2 = itsubpos2;
01076                 std::advance( itsublen2, isubst->m_subst_string.m_rlength );
01077 
01078                 if( first )
01079                     str.replace( itstrpos, itstrlen, itsubpos2, itsublen2 );
01080                 else
01081                     str.insert( itstrpos, itsubpos2, itsublen2 );
01082                 sublen = std::distance( itsubpos2, itsublen2 );
01083                 break;
01084 
01085             case subst_node::SUBST_BACKREF:
01086                 switch( isubst->m_subst_backref )
01087                 {
01088                 case subst_node::PREMATCH:
01089                     itsubpos1 = results.backref_str().begin();
01090                     itsublen1 = itsubpos1;
01091                     std::advance( itsublen1, sublen = results.rstart() );
01092                     break;
01093                 case subst_node::POSTMATCH:
01094                     itsubpos1 = results.backref_str().begin();
01095                     std::advance( itsubpos1, results.rstart() + results.rlength() );
01096                     itsublen1 = results.backref_str().end();
01097                     break;
01098                 default:
01099                     itsubpos1 = results.backref_str().begin();
01100                     std::advance( itsubpos1, results.rstart( isubst->m_subst_backref ) );
01101                     itsublen1 = itsubpos1;
01102                     std::advance( itsublen1, results.rlength( isubst->m_subst_backref ) );
01103                     break;
01104                 }
01105 
01106                 if( first )
01107                     str.replace( itstrpos, itstrlen, itsubpos1, itsublen1 );
01108                 else
01109                     str.insert( itstrpos, itsubpos1, itsublen1 );
01110                 sublen = std::distance( itsubpos1, itsublen1 );
01111                 break;
01112 
01113             case subst_node::SUBST_OP:
01114                 switch( isubst->m_op )
01115                 {
01116                 case subst_node::UPPER_ON:
01117                     rest = UPPER;
01118                     break;
01119                 case subst_node::UPPER_NEXT:
01120                     next = UPPER;
01121                     break;
01122                 case subst_node::LOWER_ON:
01123                     rest = LOWER;
01124                     break;
01125                 case subst_node::LOWER_NEXT:
01126                     next = LOWER;
01127                     break;
01128                 case subst_node::ALL_OFF:
01129                     rest = NIL;
01130                     break;
01131                 default:
01132                     REGEX_ASSERT(false);
01133                     break;
01134                 }
01135                 continue; // jump to the next item in the list
01136 
01137             default:
01138                 REGEX_ASSERT(false);
01139                 break;
01140             }
01141 
01142             first = false;
01143 
01144             // Are we upper- or lower-casing this string?
01145             if( rest )
01146             {
01147                 typename std::basic_string<CharT, TraitsT, AllocT>::iterator ibegin = str.begin();
01148                 std::advance( ibegin, strpos );
01149                 typename std::basic_string<CharT, TraitsT, AllocT>::const_iterator iend = ibegin;
01150                 std::advance( iend, sublen );
01151                 switch( rest )
01152                 {
01153                 case UPPER:
01154                     regex_toupper( ibegin, iend );
01155                     break;
01156                 case LOWER:
01157                     regex_tolower( ibegin, iend );
01158                     break;
01159                 default:
01160                     REGEX_ASSERT(false);
01161                     break;
01162                 }
01163             }
01164 
01165             // Are we upper- or lower-casing the next character?
01166             if( next )
01167             {
01168                 switch( next )
01169                 {
01170                 case UPPER:
01171                     traits_type::assign( str[strpos], regex_toupper( str[strpos] ) );
01172                     break;
01173                 case LOWER:
01174                     traits_type::assign( str[strpos], regex_tolower( str[strpos] ) );
01175                     break;
01176                 default:
01177                     REGEX_ASSERT(false);
01178                     break;
01179                 }
01180                 next = NIL;
01181             }
01182 
01183             strpos += sublen;
01184         }
01185 
01186         // If *first* is still true, then we never called str.replace, and the substitution
01187         // string is empty. Erase the part of the string that the pattern matched.
01188         if( first )
01189             str.erase( strpos, strlen );
01190 
01191         // return length of the substitution
01192         return strpos - old_strpos;
01193     }

01195     template< typename CharT, typename TraitsT, typename AllocT >
01196     static size_t _do_subst
01197     (
01198         rpattern_type const & pat,
01199         std::basic_string<CharT, TraitsT, AllocT> & str,
01200         basic_subst_results<CharT, TraitsT, AllocT> & results,
01201         size_type pos,
01202         size_type len
01203     )
01204     {
01205         typedef std::basic_string<CharT, TraitsT, AllocT> string_type;
01206         typedef typename basic_subst_results<CharT, TraitsT, AllocT>::backref_vector backref_vector;
01207 
01208         results.m_pbackref_str = pat._save_backrefs() ? &( results.m_backref_str = str ) : &str;
01209         results.m_ibegin = results.m_pbackref_str->begin();
01210 
01211         size_t csubst = 0;
01212         size_type stop_offset = results.m_pbackref_str->size();
01213         if( len != rpattern_type::npos )
01214             stop_offset = regex_min( size_t( pos + len ), stop_offset );
01215 
01216         match_param<IterT> param( results.m_ibegin, results.m_ibegin, results.m_ibegin, 0, 0 );
01217 
01218         std::advance( param.m_imatchbegin, pos );
01219         std::advance( param.m_iend, stop_offset );
01220         param.m_ibufferbegin = param.m_imatchbegin;
01221 
01222         if( GLOBAL & pat.flags() )
01223         {
01224             bool const fAll   = ( ALLBACKREFS   == ( ALLBACKREFS   & pat.flags() ) );
01225             bool const fFirst = ( FIRSTBACKREFS == ( FIRSTBACKREFS & pat.flags() ) );
01226             backref_vector rgtempbackrefs( results.m_rgbackrefs.get_allocator() ); // temporary vector used if fsave_backrefs
01227 
01228             size_type pos_offset = 0; // keep track of how much the backref_str and
01229                                       // the current string are out of sync
01230 
01231             while( _do_try_match( pat, param, results.m_rgbackrefs, false ) )
01232             {
01233                 backref_type const & br = param.m_prgbackrefs[0];
01234                 ++csubst;
01235                 size_type match_length = std::distance( br.first, br.second );
01236                 pos = std::distance( results.m_ibegin, br.first );
01237                 size_type subst_length = _do_subst_internal( str, results, pat, pos + pos_offset, match_length );
01238 
01239                 if( pat._save_backrefs() )
01240                 {
01241                     pos += match_length;
01242                     pos_offset += ( subst_length - match_length );
01243 
01244                     // Handle specially the backref flags
01245                     if( fFirst )
01246                     {
01247                         rgtempbackrefs.push_back( br );
01248                     }
01249                     else if( fAll )
01250                     {
01251                         rgtempbackrefs.insert(
01252                             rgtempbackrefs.end(),
01253                             results.m_rgbackrefs.begin(),
01254                             results.m_rgbackrefs.end() );
01255                     }
01256                     else
01257                     {
01258                         rgtempbackrefs.swap( results.m_rgbackrefs );
01259                     }
01260                 }
01261                 else
01262                 {
01263                     pos += subst_length;
01264                     stop_offset += ( subst_length - match_length );
01265                     results.m_ibegin = results.m_pbackref_str->begin();
01266 
01267                     // we're not saving backref information, so we don't
01268                     // need to do any special backref maintenance here
01269                 }
01270 
01271                 // prevent a pattern that matches 0 characters from matching
01272                 // again at the same point in the string
01273                 param.m_no0len = ( 0 == match_length );

01350 //
01351 // Function:    process_escapes
01352 //
01353 // Description: Turn the escape sequnces \f \n \r \t \v \\ into their
01354 //              ASCII character equivalents. Also, optionally process
01355 //              perl escape sequences.
01356 //
01357 // Returns:     void
01358 //
01359 // Arguments:   str      - the string to process
01360 //              fPattern - true if the string is to be processed as a regex
01361 //
01362 // Notes:       When fPattern is true, the perl escape sequences are not
01363 //              processed.  If there is an octal or hex excape sequence, we
01364 //              don't want to turn it into a regex metacharacter here.  We
01365 //              leave it unescaped so the regex parser correctly interprests
01366 //              it as a character literal.
01367 //
01368 // History:     8/1/2001 - ericne - Created
01369 //
01370 // --------------------------------------------------------------------------
01371 template< typename CharT, typename TraitsT, typename AllocT >
01372 inline void process_escapes( std::basic_string<CharT, TraitsT, AllocT> & str, bool fPattern ) //throw()
01373 {
01374     typedef typename std::basic_string<CharT, TraitsT, AllocT>::size_type size_type;
01375 
01376     size_type i = 0;
01377     size_type const npos = std::basic_string<CharT, TraitsT, AllocT>::npos;
01378 
01379     if( str.empty() )
01380         return;
01381 
01382     while( npos != ( i = str.find( REGEX_CHAR(CharT,'\\'), i ) ) )
01383     {
01384         if( str.size() - 1 == i )
01385             return;
01386 
01387         switch( str[i+1] )
01388         {
01389         case REGEX_CHAR(CharT,'a'):
01390             str.replace( i, 2, 1, REGEX_CHAR(CharT,'\a') );
01391             break;
01392         case REGEX_CHAR(CharT,'b'):
01393             if( ! fPattern )
01394                 str.replace( i, 2, 1, REGEX_CHAR(CharT,'\b') );
01395             else
01396                 ++i;
01397             break;
01398         case REGEX_CHAR(CharT,'e'):
01399             str.replace( i, 2, 1, CharT( 27 ) );
01400             break;
01401         case REGEX_CHAR(CharT,'f'):
01402             str.replace( i, 2, 1, REGEX_CHAR(CharT,'\f') );
01403             break;
01404         case REGEX_CHAR(CharT,'n'):
01405             str.replace( i, 2, 1, REGEX_CHAR(CharT,'\n') );
01406             break;
01407         case REGEX_CHAR(CharT,'r'):
01408             str.replace( i, 2, 1, REGEX_CHAR(CharT,'\r') );
01409             break;
01410         case REGEX_CHAR(CharT,'t'):
01411             str.replace( i, 2, 1, REGEX_CHAR(CharT,'\t') );
01412             break;
01413         case REGEX_CHAR(CharT,'v'):
01414             str.replace( i, 2, 1, REGEX_CHAR(CharT,'\v') );
01415             break;
01416         case REGEX_CHAR(CharT,'\\'):
01417             if( fPattern )
01418             {
01419                 if( i+3 < str.size() && REGEX_CHAR(CharT,'\\') == str[i+2] && REGEX_CHAR(CharT,'\\') == str[i+3] )
01420                     str.erase( i, 2 );
01421                 ++i;
01422             }
01423             else
01424                 str.erase( i, 1 );
01425             break;
01426         case REGEX_CHAR(CharT,'0'): case REGEX_CHAR(CharT,'1'): case REGEX_CHAR(CharT,'2'): case REGEX_CHAR(CharT,'3'):
01427         case REGEX_CHAR(CharT,'4'): case REGEX_CHAR(CharT,'5'): case REGEX_CHAR(CharT,'6'): case REGEX_CHAR(CharT,'7'):
01428             if( ! fPattern )
01429             {
01430                 size_t j=i+2;
01431                 CharT ch = CharT( str[i+1] - REGEX_CHAR(CharT,'0') );
01432                 for( ; j-i < 4 && j < str.size() && REGEX_CHAR(CharT,'0') <= str[j] && REGEX_CHAR(CharT,'7') >= str[j]; ++j )
01433                     ch = CharT( ch * 8 + ( str[j] - REGEX_CHAR(CharT,'0') ) );
01434                 str.replace( i, j-i, 1, ch );
01435             }
01436             break;
01437         case REGEX_CHAR(CharT,'x'):
01438             if( ! fPattern )
01439             {
01440                 CharT ch = 0;
01441                 size_t j=i+2;
01442                 for( ; j-i < 4 && j < str.size() && detail::regex_isxdigit( str[j] ); ++j )
01443                     ch = CharT( ch * 16 + detail::regex_xdigit2int( str[j] ) );
01444                 str.replace( i, j-i, 1, ch );
01445             }
01446             break;
01447         case REGEX_CHAR(CharT,'c'):
01448             if( ! fPattern && i+2 < str.size() )
01449             {
01450                 CharT ch = str[i+2];
01451                 if( REGEX_CHAR(CharT,'a') <= ch && REGEX_CHAR(CharT,'z') >= ch )
01452                     ch = detail::regex_toupper( ch );
01453                 str.replace( i, 3, 1, CharT( ch ^ 0x40 ) );
01454             }
01455             break;
01456         default:
01457             if( fPattern )
01458                 ++i;
01459             else
01460                 str.erase( i, 1 );
01461             break;
01462         }
01463         ++i;
01464         if( str.size() <= i )
01465             return;
01466     }
