00511     const_iterator eat_whitespace( iterator & icur, const_iterator iend )
00512     {
00513         if( m_flags & EXTENDED )
00514         {
00515             while( iend != icur && ( REGEX_CHAR(CharT,'#') == *icur || detail::regex_isspace( *icur ) ) )
00516             {
00517                 if( REGEX_CHAR(CharT,'#') == *icur++ )
00518                 {
00519                     while( iend != icur && REGEX_CHAR(CharT,'\n') != *icur++ ) {}
00520                 }
00521                 else
00522                 {
00523                     for( ; iend != icur && detail::regex_isspace( *icur ); ++icur ) {}
00524                 }
00525             }
00526         }

00557     TOKEN reg_token( iterator & icur, const_iterator iend )
00558     {
00559         REGEX_ASSERT( iend != icur );
00560         if( iend == eat_whitespace( icur, iend ) )
00561             return NO_TOKEN;
00562         TOKEN tok = look_up( *icur, s_rgreg );
00563         if( tok )
00564             ++icur;
00565         if( ESCAPE == tok && iend != icur )
00566         {
00567             tok = look_up( *icur, s_rgescape );
00568             if( tok )
00569                 ++icur;
00570             else
00571                 tok = ESCAPE;
00572         }
00573         return tok;
00574     }
00575     TOKEN quant_token( iterator & icur, const_iterator iend )
00576     {
00577         REGEX_ASSERT( iend != icur );
00578         if( iend == eat_whitespace( icur, iend ) )
00579             return NO_TOKEN;
00580         TOKEN tok = NO_TOKEN;
00581         switch( *icur )
00582         {
00583         case REGEX_CHAR(CharT,'*'):
00584             tok = min_quant( icur, iend ) ? ZERO_OR_MORE_MIN : ZERO_OR_MORE;
00585             break;
00586         case REGEX_CHAR(CharT,'+'):
00587             tok = min_quant( icur, iend ) ? ONE_OR_MORE_MIN : ONE_OR_MORE;
00588             break;
00589         case REGEX_CHAR(CharT,'?'):
00590             tok = min_quant( icur, iend ) ? ZERO_OR_ONE_MIN : ZERO_OR_ONE;
00591             break;
00592         case REGEX_CHAR(CharT,'}'):
00593             tok = min_quant( icur, iend ) ? END_RANGE_MIN : END_RANGE;
00594             break;
00595         case REGEX_CHAR(CharT,'{'):
00596             tok = BEGIN_RANGE;
00597             ++icur;
00598             break;

00600             tok = RANGE_SEPARATOR;
